- 0.328 \times \mbox{h\_dist}
+ 0.140 \times \mbox{h\_f\_b\_dist}
- 0.100 \times \mbox{m\_avg\_w} \nonumber \\
+ 0.019 \times \mbox{m\_sim}
+ 0.000 \times \mbox{first\_second}
- 0.127 \times \mbox{sigma}
+ 3.610
\end{align}
